# Bootstrap vzorkovn ze simulace na velikost relnho datasetu
n_real = len(df_real)

sim_samples = []

for _ in range(1000):
    sample = df_sim.sample(n=n_real, replace=True, random_state=122 + _)
    sim_samples.append(sample['total_time'])

KS test pro kad podvzorkovn

In[10]:
# Vpoet KS p-hodnot pro porovnn relnch a simulovanch dat
p_vals = []

for s in sim_samples:
    _, p = ks_2samp(df_real['total_time'], s)
    p_vals.append(p)

Cohens d (velikost efektu)

In[19]:
mean_diff = abs(df_real['total_time'].mean() - df_sim['total_time'].mean())

pooled_std = np.sqrt(
    (df_real['total_time'].std()**2 + df_sim['total_time'].std()**2) / 2
)

d = mean_diff / pooled_std

print("Cohen's d:", d)

Cohen's d: 0.18425210268660908

Cohen's d < 0.2 zanedbateln vznam
