# nastaven formtu vpisu pro zobrazen a 8 desetinnch mst.

# Definice prmrnch dob trvn jednotlivch stav systmu a intenzit pechodu do neprovoznch stav na zklad experimentlnch dat
dt = 1  # asov krok (s)

T = np.array([1, 1, 4, 2, 3, 13, 4, 13, 600, 120, 3000, 1000])

lam9  = 1/20500
lam10 = 1/820
lam11 = 1/144000
lam12 = 1/14400
In[4]:
# Inicializace pechodov matice Markovskho etzce
P = np.zeros((12,12))

Vpoet pechod pro provozn stavy (S1S7)

In[5]:
# Vpoet pechodovch pravdpodobnost mezi provoznmi stavy s vyuitm diskrtn aproximace CTMC.
# V ppad zporn pravdpodobnosti setrvn ve stavu je pouita kompetitivn normalizace pechod
for i in range(7):

    mu = 1/T[i]
    total = mu + lam9 + lam10 + lam11 + lam12

    # --- CTMC aproximace ---
    P[i, i+1] = mu * dt
    P[i, 8]   = lam9  * dt
    P[i, 9]   = lam10 * dt
    P[i,10]   = lam11 * dt
    P[i,11]   = lam12 * dt

    Pii = 1 - total * dt

    # --- Kontrola zporn pravdpodobnosti ---
    if Pii < 0:
        P[i, i+1] = mu / total
        P[i, 8]   = lam9  / total
        P[i, 9]   = lam10 / total
        P[i,10]   = lam11 / total
        P[i,11]   = lam12 / total
        P[i, i]   = 0
    else:
        P[i, i] = Pii

Pechody z neprovoznch stav (S9S12)

In[7]:
# Modelovn nvratu systmu z neprovoznch stav zpt do potenho provoznho stavu S1
for i in range(8,12):
    mu = 1/T[i]
    P[i,0] = mu * dt
    P[i,i] = 1 - mu * dt

# Vpoet stacionrnho rozdlen pravdpodobnost pomoc vlastnho een matice.
eigvals, eigvecs = np.linalg.eig(P.T)

idx = np.argmin(np.abs(eigvals-1))

pi = eigvecs[:,idx]
pi = np.real(pi / np.sum(pi))

print("\nStacionrn rozdlen pi:")
print(np.round(pi,6))
