In[5]:
n_states = 12

# poet iteranch krok (sekund)
n_iter = 100_000

# smrodatn odchylka nvrhovho normlnho rozdlen
sigma = 1.5

samples = np.zeros(n_iter, dtype=int)

# poten stav S1 (index 0)
current_state = 0
samples[0] = current_state

# nastaven seedu (potenho stavu genertoru nhodnch sel)
np.random.seed(122)

MetropolisHastings simulace

In[6]:
for t in range(1, n_iter):

    # --- nvrh novho stavu ---
    proposal = int(np.round(
        np.random.normal(loc=current_state, scale=sigma)
    ))

    # omezen na interval stav
    proposal = np.clip(proposal, 0, n_states-1)

    # --- pravdpodobnosti ---
    p_current = pi[current_state]
    p_proposal = pi[proposal]

    # symetrick nvrhov rozdlen  zkrcen pomru
    alpha = min(1, p_proposal / p_current)

    # --- pijet / zamtnut ---
    if np.random.rand() < alpha:
        current_state = proposal

    samples[t] = current_state
