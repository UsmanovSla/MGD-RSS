In[4]:
n_states = 12

# poet iteranch krok (sekund)
n_iter = 1_000_000

# nastaven seedu (potenho stavu genertoru nhodnch sel)
rng = np.random.default_rng(seed=122)

samples = np.zeros(n_iter, dtype=int)

# poten stav S1 (index 0)
current_state = 0
samples[0] = current_state

Pm vzorkovn z pechodov matice

In[5]:
for t in range(1, n_iter):

    current_state = rng.choice(
        n_states,
        p=P[current_state]
    )

    samples[t] = current_state
