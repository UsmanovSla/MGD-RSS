# SPECIFIKACE ROBOTICKHO SYSTMU

speed_load = 0.5      # m/s, nastaven rychlost TCP pi manipulaci s prvkem (se zt)
speed_unload = 1.0    # m/s, nastaven rychlost TCP bez zte
accel = 1                 # m/s^2, nastaven zrychlen TCP

# vpoet asu potebnho k dosaen nastaven rychlosti
time_to_load_speed = speed_load / accel
dist_to_load_speed = (1/2) * accel * time_to_load_speed  # uraen drha pi akceleraci

# vpoet asu potebnho k dosaen 0 rychlosti
time_to_unload_speed = speed_unload / accel
dist_to_unload_speed = (1/2) * accel * time_to_unload_speed  # uraen drha pi deakceleraci

# pracovn dosah robotickho ramene
range_max = 3_904         # mm, maximln pracovn dosah
range_min = 1_912         # mm, minimln pracovn dosah (orientan)

# SPECIFIKACE TECHNOLOGICKHO PROCESU ZDN

time_refer_2_refer = 20   # s, prmrn doba pohybu z referennho bodu k referennmu bodu (depaletizace, verifikace)
time_mounting = 3         # s, doba manipulaci v clov poloze (umstn prvku)
brick_thickness = 440     # mm, tlouka zdicho prvku (Porotherm 440 Profi)
brick_height = 250        # mm, vka zdicho prvku (Porotherm 440 Profi)
brick_width = 250         # mm, ka zdicho prvku (Porotherm 440 Profi)

# SOUADNICE REFERENNHO BODU (nad verifikanm stolem)

refer_x = 2_000           # mm, souadnice X referennho bodu
refer_y = 3_500           # mm, souadnice Y referennho bodu
refer_z = 1_000           # mm, souadnice Z referennho bodu

# POLOHA ROBOTICKHO RAMENE

robot_x = 2_250           # mm, souadnice X zkladny robotu
robot_y = 2_420           # mm, souadnice Y zkladny robotu
robot_z = 1_000           # mm, souadnice Z zkladny robotu

Definice funkce pro vpoet celkov doby pracovnho cyklu bez stochstickch vliv

def calculation_dist(x, y, z):
    """
    Funkce pro vpoet drhy trajektorie od referennho bodu k clov poloze prvku

    Parametry:
    x, y, z (int): souadnice clov polohy prvku [mm]

    Nvratov hodnota:
    dist (int): drha trajektorie od referennho bodu k clov poloze prvku [mm]
    """
    dist = 0

    # Fze 1: drha od referennho bodu k clov stn.
    dist = ((refer_z - (z + brick_height*2))**2 + (refer_x - brick_thickness//2)**2)**(1/2)

    # Fze 2: drha ve smru osy X
    if x != brick_thickness / 2:
        dist = dist + abs(x - brick_thickness//2 + brick_width * 2)

    # Fze 3: drha ve smru osy Y
    if y != brick_thickness / 2:
        dist = dist + abs(y - refer_y + brick_width * 2)

    return int(dist)


def simulate_time(dist):
    """
    Funkce pro vpoet celkov doby pracovnho cyklu robotickho zdn.

    Parametry:
    dist (int): drha trajektorie od referennho bodu k clov poloze prvku [mm]

    Nvratov hodnota:
    total_time (int): celkov doba pracovnho cyklu [s]
    """
    # pevn technologick asy (manipulace a pesuny mezi pevnmi body)
    total_time = time_refer_2_refer

    # pohyb s naloenm prvkem (pevod mm  m)
    total_time += (dist - dist_to_load_speed) / 1_000 / speed_load

    # manipulace v clov poloze
    total_time += time_mounting

    # pohyb bez zte (zptn pohyb)
    total_time += (dist - dist_to_unload_speed) / 1_000 / speed_unload

    # zapoten akceleranch as
    total_time += time_to_load_speed + time_to_unload_speed
    return total_time
