# SPECIFIKACE ROBOTICKHO SYSTMU

speed_max_load = 0.5      # m/s, maximln rychlost TCP pi manipulaci s prvkem (se zt)
speed_max_unload = 1.0    # m/s, maximln rychlost TCP bez zte
accel = 1                 # m/s^2, zrychlen TCP

# vpoet asu potebnho k dosaen maximln rychlosti
time_to_max_load_speed = speed_max_load / accel
dist_to_max_load_speed = (1/2) * accel * time_to_max_load_speed  # uraen drha pi akceleraci

# vpoet asu potebnho k dosaen 0 rychlosti
time_to_max_unload_speed = speed_max_unload / accel
dist_to_max_unload_speed = (1/2) * accel * time_to_max_unload_speed  # uraen drha pi deakceleraci

# pracovn dosah robotickho ramene
range_max = 3_904         # mm, maximln pracovn dosah
range_min = 1_912         # mm, minimln pracovn dosah (orientan)

# SPECIFIKACE TECHNOLOGICKHO PROCESU ZDN

time_suck_on = 1.5        # s, doba pist zdicho prvku
time_suck_off = 0.5       # s, doba odst (uvolnn) prvku
time_verif_refer = 3.5    # s, doba pohybu z msta oven do referennho bodu
time_refer_pallete = 2.1  # s, doba pohybu z referennho bodu k palet

# SOUADNICE REFERENNHO BODU

refer_x = 500             # mm, souadnice X referennho bodu
refer_y = 2_450           # mm, souadnice Y referennho bodu
refer_z = 1_500           # mm, souadnice Z referennho bodu

# POLOHA ROBOTICKHO RAMENE

robot_x = 2_250           # mm, souadnice X zkladny robotu
robot_y = 2_420           # mm, souadnice Y zkladny robotu
robot_z = 1_000           # mm, souadnice Z zkladny robotu

# PARAMETRY SIMULACE

grid_simulation = 100     # mm, krok (rozlien) simulan mky

Definice funkce pro vpoet celkov doby pracovnho cyklu

def simulate_time(dist):
    """
    Funkce pro vpoet celkov doby pracovnho cyklu robotickho zdn.

    Parametry:
    dist (float): vzdlenost mezi referennm bodem a clovou polohou prvku [mm]

    Nvratov hodnota:
    total_time (float): celkov doba pracovnho cyklu [s]
    """
    # pevn technologick asy (manipulace a pesuny mezi pevnmi body)
    total_time = time_suck_on + time_pallete_verif
    total_time += time_suck_off + time_suck_on + time_verif_refer

    # pohyb s naloenm prvkem (pevod mm  m)
    total_time += (dist - dist_to_max_load_speed) / 1_000 / speed_max_load

    # manipulace v clov poloze a nvrat
    total_time += time_suck_off + time_refer_pallete

    # pohyb bez zte (zptn pohyb)
    total_time += (dist - dist_to_max_unload_speed) / 1_000 / speed_max_unload

    # zapoten akceleranch as
    total_time += time_to_max_load_speed + time_to_max_unload_speed
    return total_time
